cmake_minimum_required(VERSION 3.27)

set(APP_NAME $name)

project(
    ${APP_NAME}
    VERSION 1.0 # app_version
    DESCRIPTION "$description"
    LANGUAGES CXX
)
set(CMAKE_CXX_STANDARD 98)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_BUILD_TYPE Release)
# set(CMAKE_BUILD_TYPE Debug)

# set variable for build directory
set(BUILD_DIR "${CMAKE_CURRENT_SOURCE_DIR}/build")
# create build directory
file(MAKE_DIRECTORY ${BUILD_DIR})

add_executable("${APP_NAME}_bin") # build executable
add_library("${APP_NAME}_shared_lib" SHARED) # build dynamic library
add_library("${APP_NAME}_static_lib") # build static library

set(sources
    main.cpp
)

# set buld directories globally
#set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${BUILD_DIR}")
#set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY "${BUILD_DIR}")
#set(CMAKE_LIBRARY_OUTPUT_DIRECTORY_RELEASE "${BUILD_DIR}")
# set_property(TARGET PROPERTY RUNTIME_OUTPUT_DIRECTORY ${BUILD_DIR})

# or per target by
set_property(TARGET "${APP_NAME}_bin" PROPERTY RUNTIME_OUTPUT_DIRECTORY ${BUILD_DIR})
set_property(TARGET "${APP_NAME}_shared_lib" PROPERTY LIBRARY_OUTPUT_DIRECTORY ${BUILD_DIR})
set_property(TARGET "${APP_NAME}_static_lib" PROPERTY ARCHIVE_OUTPUT_DIRECTORY ${BUILD_DIR})

target_sources("${APP_NAME}_bin" PRIVATE ${sources})
target_sources("${APP_NAME}_shared_lib" PRIVATE ${sources})
target_sources("${APP_NAME}_static_lib" PRIVATE ${sources})
